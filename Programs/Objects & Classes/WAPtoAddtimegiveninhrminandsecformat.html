#include <iostream>
using namespace std;

class Time
{
    int hr, min, sec;

public:
    void input()
    {
        cout << "Enter the time in hr,min,sec: ";
        cin >> hr >> min >> sec;
    }

    Time add(Time t2)
    {
        Time temp;
        temp.sec = sec + t2.sec;
        temp.min = min + t2.min + temp.sec / 60;
        temp.sec = temp.sec % 60;
        temp.hr = hr + t2.hr + temp.min / 60;
        temp.min = temp.min % 60;
        return temp;
    }
    void display()
    {
        cout << "The time after addition is: ";
        cout << hr << " hr\t" << min << " min\t" << sec << " sec";
    }
};

int main()
{
    system("CLS");
    class Time t1, t2, t3;
    t1.input();
    t2.input();
    t3 = t1.add(t2);
    t3.display();
}
