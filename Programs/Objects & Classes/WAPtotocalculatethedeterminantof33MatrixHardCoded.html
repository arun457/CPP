// WAP to to calculate the determinant of 3*3 Matrix.  [HardCoded]

#include <iostream>
using namespace std;

int main()
{
    system("CLS");
    int a[3][3], i, j, det;

    cout << "Enter the Elements of Matrix:" << endl;
    for (i = 0; i < 3; i++)
    {
        for (j = 0; j < 3; j++)
        {
            cin >> a[i][j];
        }
    }

    det = a[0][0] * ((a[1][1] * a[2][2]) - (a[2][1] * a[1][2])) - a[0][1] * (a[1][0] * a[2][2] - a[2][0] * a[1][2]) + a[0][2] * (a[1][0] * a[2][1] - a[2][0] * a[1][1]);

    cout << "Determinant of Matrix is: " << det;
    return 0;
}
