#include <iostream>
using namespace std;

bool can_proceed = true;
static int timeindex = 1;

class Time
{
    int year, month, day;

public:
    void input()
    {
        cout << "Enter the " << timeindex << "th time in Year, Month, Day format: ";
        cin >> year >> month >> day;
        timeindex = timeindex + 1;
    }

    void check_validation(Time t)
    {
        int day_of_t1 = (year * 365) + (month * 30) + day;
        int day_of_t2 = (t.year * 365) + (t.month * 30) + t.day;

        if (day_of_t1 < day_of_t2)
        {
            cout << "You have entered t1<t2" << endl
                 << "Please Enter t1>t2" << endl;
            can_proceed = false;
        }
    }

    Time subtract(Time t)
    {
        Time temp;
        if (day < t.day)
        {
            day = day + 30;
            month = month - 1;
        }
        temp.day = day - t.day;
        if (month < t.month)
        {
            month = month + 12;
            year = year - 1;
        }
        temp.month = month - t.month;
        temp.year = year - t.year;

        return temp;
    }

    void display_difference()
    {
        cout << "The Difference of two time is: " << year << " Year\t" << month << " Month\t" << day << " day" << endl;
    }
};

int main()
{
    system("CLS");
    Time t1, t2, t3;
    cout << "Enter the time (t1>t2) " << endl;
    t1.input();
    t2.input();
    t1.check_validation(t2);

    if (can_proceed != false)
    {
        t3 = t1.subtract(t2);
        t3.display_difference();
    }
    else
    {
        cout << "Exitting the Program";
    }
    return 0;
}
