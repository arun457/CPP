// WAP designing a class called midpoint to find midpoint between two points by returning object
//  from member funtion using this pointer

#include <iostream>
using namespace std;

class midpoint
{
    float x, y, z;

public:
    midpoint() // for initializing temp object;
    {
    }

    midpoint(float a, float b, float c) // for initializing p1 and p2
    {
        x = a;
        y = b;
        z = c;
    }

    midpoint find_midpoint(midpoint p) // finding midpoint
    {
        midpoint temp;
        temp.x = (this->x + p.x) / 2;
        temp.y = (this->y + p.y) / 2;
        temp.z = (this->z + p.z) / 2;
        return temp;
    }

    void display() // display fxn
    {
        cout << "Midpoint = " << x << " , " << y << " , " << z;
    }
};

int main()
{
    system("CLS");

    // creating object
    midpoint p1(2, 0, 0), p2(4, 0, 0);
    midpoint p3 = p1.find_midpoint(p2); // calling find_midpoint

    // displaying
    p3.display();
    return 0;
}
