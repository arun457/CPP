// A friend function is not a member of any class but has full access to the 
//members of class where it is declares as friend.

#include <iostream>
using namespace std;

class Time
{
    int hr;

public:
    friend void set_data(Time &t1);
};

void set_data(Time &t1)
{
    t1.hr = 2; // Accessing the private member of an object.
    cout << "The hour of t is " << t1.hr << " hr";
}

int main()
{
    system("CLS");
    Time t;
    set_data(t); // calling Friend function without any object.
    return 0;
}
