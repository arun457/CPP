// WAP a meaningful program that shows the use of constant object and
// constant function along with the use of const cast operator.

#include <iostream>
using namespace std;

class Facebook
{
    string password = "old password";
    string &password_reference = password;

public:
    void request_password_change() const
    {
        int choice;

        cout << "Press 1 to change your password." << endl;
        cout << "Press 2 to cancel the operation." << endl;

        cin >> choice;

        switch (choice)
        {
        case 1:
            change_password();
            break;

        case 2:
            cout << "Operation Cancelled.";
            exit(0);
            break;

        default:
            cout << "Invalid input";
            exit(0);
            break;
        }
    }

    void change_password() const
    {
        string new_password;
        cout << "Enter the new Password" << endl;
        cin >> new_password;
        string &second_reference = const_cast<string &>(password_reference); // removing constantness using const cast
        second_reference = new_password;                                     // changing password
    }

    void Password_display() const
    {
        cout << "password = " << password << endl;
    }
};

int main()
{
    system("CLS");

    const Facebook user1;
    user1.Password_display();
    user1.request_password_change();
    user1.Password_display();

    return 0;
}

        {
            cout << "Operation Cancelled";
            exit(0);
        }
