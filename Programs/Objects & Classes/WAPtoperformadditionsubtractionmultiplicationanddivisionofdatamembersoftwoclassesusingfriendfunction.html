// WAP to perform addition, subtraction, multiplication and division
// of data members of two classes using friend function.

#include <iostream>
using namespace std;

class Arun2;

class Arun1
{
    int x;

public:
    Arun1()
    {
        x = 3;
    }

    friend void friend_function(Arun1 &A1, Arun2 &A2);
};

class Arun2
{
    int x;

public:
    Arun2()
    {
        x = 2;
    }

    friend void friend_function(Arun1 &A1, Arun2 &A2);
};

void friend_function(Arun1 &A1, Arun2 &A2)
{
    cout << "The sum of X of A1 and A2 is: " << A1.x + A2.x << endl;
    cout << "The Difference of X of A1 and A2 is: " << A1.x - A2.x << endl;
    cout << "The Product of X of A1 and A2 is: " << A1.x * A2.x << endl;
    cout << "The Division of X of A1 and A2 is: " << (float)A1.x / A2.x << endl;
}

int main()
{
    system("CLS");
    Arun1 A1;
    Arun2 A2;
    friend_function(A1, A2);
    return 0;
}
