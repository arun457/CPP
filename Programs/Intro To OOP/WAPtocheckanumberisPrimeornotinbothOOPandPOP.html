//WAP to check a number is Prime or not in both OOP and POP.

#include <stdio.h>
#include <conio.h>
#include <stdlib.h>

void main()
{
    system("cls");
    int i = 0, n, prime = 1;
    printf("Enter the number to check: ");
    scanf("%d", &n);

    for (i = 2; i <= (n / 2); i++)
    {
        if (n % i == 0)
        {
            prime = 0;
            break;
        }
    }

    if (prime == 0)
    {
        printf("The given number is not Prime.");
    }
    else
    {
        printf("The given number is Prime");
    }
}
