//WAP with class Employee, data member e_id and e_name. Make input,display function to input and display data in console 
//and read and write function to read and write from and to file.

#include <iostream>
#include <fstream>
using namespace std;

class Employee
{
    int e_id;
    char e_name[20];

public:
    void input()
    {
        cout << "Enter the Employee ID: ";
        cin >> e_id;
        cout << "Enter the Employee Name: ";
        fflush(stdin); // flushing the memory to flush the entered \n else getline will read that hit enter as input and skips.
        cin.getline(e_name, 20);
    }

    void display()
    {
        cout << "Employee ID: " << e_id << endl;
        cout << "Employee Name: " << e_name;
    }

    void write()
    {
        ofstream fout("Employee.txt");            // output stream to write to file
        char *c = reinterpret_cast<char *>(this); // converting from object to pointer
        fout.write(c, sizeof(*this));             // writing the info of object e to the file
        fout.close();                             // closing the file
    }

    void read()
    {
        ifstream fin("Employee.txt");          // input stream to read from the file
        fin.read((char *)this, sizeof(*this)); // reading from the file --(char *)&e1 is equivalent reinterpret cast
        fin.close();
    }
};

int main()
{
    system("CLS");

    Employee e, e1;
    cout << "Enter Employee Details: \n";

    e.input();
    e.write();

    e1.read();
    cout << "\nEmployee Details: " << endl;
    e1.display();

    return 0;
}
