#include <iostream>
#include <fstream>
#include <conio.h>
using namespace std;

class student
{
    int roll;
    char name[40];

    int telephone;
    int score;

public:
    void getdata()
    {
        cout << "\nEnter name of Student: ";
        cin.getline(name, 40);
        cout << "Enter roll number: ";
        cin >> roll;
        cout << "Enter phone number: ";
        cin >> telephone;
        cout << "Enter score: ";
        cin >> score;
    }

    void showdata()
    {
        cout << "\nName of Student: " << name;
        cout << "\nRoll Number: " << roll;
        cout << "\nPhone Number: " << telephone;
        cout << "\nScore: " << score;
    }
};

void input_to_file()
{
    fstream file;
    student s1;
    file.open("student.txt");
    cout << "\nEnter Student data";
    s1.getdata();
    file.write((char *)&s1, sizeof(s1));
}

void output_from_file()
{
    fstream fout;
    student s2;
    fout.open("student.txt");
    fout.seekg(0);
    fout.read((char *)&s2, sizeof(s2));
    while (!fout.eof())
    {
        cout << "\nStudent record: ";
        s2.showdata();
        fout.read((char *)&s2, sizeof(s2));
    }
}

int main()
{
    system("CLS");

    input_to_file();
    output_from_file();

    return 0;
}
