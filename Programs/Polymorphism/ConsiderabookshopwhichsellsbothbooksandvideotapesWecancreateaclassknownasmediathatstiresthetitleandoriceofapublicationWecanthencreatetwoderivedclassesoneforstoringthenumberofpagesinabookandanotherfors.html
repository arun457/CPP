// Consider a book shop which sells both books and video-tapes.
// We can create a class known as media that stires the title and orice of a publication.
// We can then create two derived classes, one for storing the number of pages in a book and another for storing the playing time of tape.

#include <iostream>
#include <string.h>
using namespace std;

class media
{
protected:
    char title[50];
    float price;

public:
    media(char *t, float p)
    {
        strcpy(title, t);
        price = p;
    }

    virtual void display()
    {
        cout << "Display of media class called." << endl;
    }
};

class Book : public media
{
    int pages;

public:
    Book(char *t, float p, int pg) : media(t, p)
    {
        pages = pg;
    }

    void display()
    {
        cout << "Title = " << title << endl
             << "Price = " << price << endl
             << "Pages = " << pages << endl;
    }
};

class Tape : public media
{
    float time;

public:
    Tape(char *t, float p, float T) : media(t, p)
    {
        time = T;
    }

    void display()
    {
        cout << "Title = " << title << endl
             << "Price = " << price << endl
             << "Play Time = " << time << endl;
    }
};

int main()
{
    system("CLS");
    media *ptr[2];
    Book book("book_title,", 500, 150);
    Tape tape("Tape_name", 600, 50);
    ptr[0] = &book;
    ptr[1] = &tape;
    ptr[0]->display();
    ptr[1]->display();
    return 0;
}
