// WAP to add subtract and multiply two matrices

#include <iostream>
#define Size 2
using namespace std;

class Matrix
{
    int a[Size][Size];

public:
    Matrix()
    {
    }

    Matrix(Matrix &m)
    {
        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                a[i][j] = m.a[i][j];
            }
        }
    }

    Matrix(int m[Size][Size])
    {
        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                a[i][j] = m[i][j];
            }
        }
    }

    Matrix operator+(Matrix m)
    {
        Matrix temp;

        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                temp.a[i][j] = a[i][j] + m.a[i][j];
            }
        }
        return temp;
    }

    Matrix operator-(Matrix m)
    {
        Matrix temp;

        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                temp.a[i][j] = a[i][j] - m.a[i][j];
            }
        }
        return temp;
    }

    Matrix operator*(Matrix m)
    {
        Matrix temp;

        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                temp.a[i][j] = 0;
                for (int k = 0; k < Size; k++)
                {
                    temp.a[i][j] = temp.a[i][j] + a[i][k] * m.a[k][j];
                }
            }
        }
        return temp;
    }

    void display()
    {
        cout << "The Resultant Matrix is: " << endl;

        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                cout << a[i][j] << "\t";
            }
            cout << endl;
        }
    }
};

int main()
{
    system("CLS");
    int mat1[Size][Size] = {1, 2, 3, 4};
    int mat2[Size][Size] = {2, 3, 4, 5};
    Matrix m1(mat1), m2(mat2), m3, m4, m5;
    m3 = m1 + m2;
    m4 = m1 - m2;
    m5 = m1 * m2;

    cout << "Addition: " << endl;
    m3.display();

    cout << "\nSubtraction: " << endl;
    m4.display();

    cout << "\nDivision: " << endl;
    m5.display();
    return 0;
}
