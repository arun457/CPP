// WAP to create class named City having two member variables CityName (char[20]) and DistFromKtm(float).
// Add member functions to set and retrieve the CityName and DistFromKtm seperately. Add operator overloading to find the distance between the
// cities (just find the difference of DistFromKtm) and sum of distances of those cities from Kathmandu.
// In the main function, initialize three city objects. Set the first and second city to be Pokhara and Dhangadi. Display the sum of DistFromKtm of
// Pokhara and Dhangadi and Distance between Pokhara and Dhangadi

#include <iostream>
#include <string.h>
using namespace std;

class City
{
    char CityName[20];
    float DistFromKtm;

public:
    City()
    {
    }

    City(char *name, float dist)
    {
        strcpy(CityName, name);
        DistFromKtm = dist;
    }

    City operator+(City c)
    {
        City temp;
        temp.DistFromKtm = DistFromKtm + c.DistFromKtm;
        return temp;
    }

    City operator-(City c)
    {
        City temp;
        temp.DistFromKtm = DistFromKtm - c.DistFromKtm;
        return temp;
    }

    void display()
    {
        cout << "City Name = " << CityName << "\t"
             << "DistFromKtm = " << DistFromKtm << endl;
    }

    void display_sum()
    {
        cout << "Sum of DistFromKtm = " << DistFromKtm << endl;
    }

    void display_difference()
    {
        cout << "Difference of DistFromKtm = " << DistFromKtm << endl;
    }
};

int main()
{
    system("CLS");
    City c1("Pokhara", 1500), c2("Dhangadi", 1100), c3;
    c3 = c1 + c2;
    c3.display_sum();
    c3 = c1 - c2;
    c3.display_difference();
    return 0;
}
