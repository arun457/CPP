// WAP to overload the relational operators to compare the length (in meters and centimeters) of two object

#include <iostream>
using namespace std;

class Distance
{
    float m, cm;

public:
    Distance(float x, float y)
    {
        m = x;
        cm = y;
    }

    bool operator==(Distance d)
    {
        if (m == d.m && cm == d.cm)
        {
            return true;
        }
        else
        {
            return false;
        }
    }

    bool operator>(Distance d)
    {
        if (m > d.m)
        {
            return true;
        }
        else if (m == d.m)
        {
            if (cm > d.cm)
            {
                return true;
            }
            else
            {
                return false;
            }
        }
        else
        {
            return false;
        }
    }
};

int main()
{
    system("CLS");
    Distance d1(10, 10), d2(10, 6);
    if (d1 == d2)
    {
        cout << "Equal";
    }
    else if (d1 > d2)
    {
        cout << "d1>d2";
    }
    else
    {
        cout << "d1<d2";
    }
    return 0;
}
// WAP to overload the relational operators to compare the length (in meters and centimeters) of two object

#include <iostream>
using namespace std;

enum relation_set
{
    e,
    gt,
    lt
};
relation_set relation;

class Distance
{
    float m, cm;

public:
    Distance(float x, float y)
    {
        m = x;
        cm = y;
    }

    int operator==(Distance d)
    {
        if (m > d.m)
        {
            relation = gt;
        }
        else if (m < d.m)
        {
            relation = lt;
        }
        else
        {
            if (cm > d.cm)
            {
                relation = gt;
            }
            else if (cm < d.cm)
            {
                relation = lt;
            }
            else
            {
                relation = lt;
            }
        }
    }
};

int main()
{
    system("CLS");
    Distance d1(10, 4), d2(10, 5);

    d1 == d2;

    switch (relation)
    {
    case gt:
        cout << "d1>d2";
        break;
    case lt:
        cout << "d1<d2";
        break;
    case e:
        cout << "d1 = d2";
        break;
    }

    return 0;
}
// WAP to overload the relational operators to compare the length (in meters and centimeters) of two object

#include <iostream>
using namespace std;

class distance1
{
    long int meter, centimeter;

public:
    distance1()
    {
        meter = 0;
        centimeter = 0;
    }

    void read()
    {
        cout << "Enter value in Meter and Centimeter: ";
        cin >> meter >> centimeter;
        centimeter += meter * 100;
    }

    int operator>(distance1 a)
    {
        if (centimeter > a.centimeter)
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }

    int operator<(distance1 a)
    {
        if (centimeter < a.centimeter)
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }
    int operator==(distance1 a)
    {
        if (centimeter == a.centimeter)
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }
    int operator!=(distance1 a)
    {
        if (centimeter != a.centimeter)
        {
            return 1;
        }
        else
        {
            return 0;
        }
    }

    void display()
    {
        cout << meter << " Meter and " << centimeter << " centimeter" << endl;
    }
};

int main()
{
    system("CLS");
    distance1 cl, c2, c3, c4;

    cl.read();
    c2.read();

    if (cl == c2)
    {
        cout << "The values are equal.";
    }

    if (cl != c2)
    {
        cout << "The values are different and ";
    }

    if (cl > c2)
    {
        cout << "1st value is greater one.";
    }

    if (cl < c2)
    {
        cout << "2nd value is greater one.";
    }

    return 0;
}
