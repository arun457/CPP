// Develop a program using a class with 3x3 matrix as data member. Overload the * Operator so as to multiply 3x3 matrices.

#include <iostream>
using namespace std;

class Matrix
{
    int m[3][3];

public:
    Matrix()
    {
        int i, j;
        for (i = 0; i < 3; i++)
        {
            for (j = 0; j < 3; j++)
            {
                m[i][j] = 0;
            }
        }
    }

    Matrix(int mat[3][3]) // initializing the matrices
    {
        int i, j;
        for (i = 0; i < 3; i++)
        {
            for (j = 0; j < 3; j++)
            {
                m[i][j] = mat[i][j];
            }
        }
    }

    Matrix operator*(Matrix x) // multiplying Matrices
    {
        Matrix temp;
        int i, j, k;
        for (i = 0; i < 3; i++)
        {
            for (j = 0; j < 3; j++)
            {
                temp.m[i][j] = 0;
                for (k = 0; k < 3; k++)
                {
                    temp.m[i][j] = temp.m[i][j] + m[i][k] * x.m[k][j]; //Multiplying Algorithm
                }
            }
        }
        return temp;
    }

    void Display() // Displaying in Matrix Form
    {
        int i, j;
        for (i = 0; i < 3; i++)
        {
            for (j = 0; j < 3; j++)
            {
                cout << m[i][j] << "\t";
            }
            cout << endl;
        }
        cout << endl;
    }
};

int main()
{
    system("CLS");

    int mat1[3][3] = {3, 12, 4, 5, 6, 8, 1, 0, 2};
    int mat2[3][3] = {7, 3, 8, 11, 9, 5, 6, 8, 4};

    Matrix m1(mat1), m2(mat2), m3;

    cout << "Matrix 1:" << endl;
    m1.Display();

    cout << "Matrix 2:" << endl;
    m2.Display();

    m3 = m1 * m2;
    cout << "The multiplied Matrix is:" << endl;
    m3.Display();

    return 0;
}
