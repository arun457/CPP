// Create a class having an array as mermber.Overload index operator([]) to input and display the elements in the array.

#include <iostream>
using namespace std;

class Array
{
    int *a;
    int size;

public:
    Array(int p[], int s)
    {
        size = s;
        a = new int[size];
        int i;
        for (i = 0; i < size; i++)
        {
            a[i] = p[i];
        }
    }

    int &operator[](int index)
    {
        if (index >= size)
        {
            cout << "Index out of bound";
            exit(0);
        }
        return a[index];
    }

    void display()
    {
        int i;
        for (i = 0; i < size; i++)
        {
            cout << a[i] << "\t";
        }
    }
};

int main()
{
    system("CLS");

    int a_ip[] = {1, 2, 3, 4};
    Array obj(a_ip, 4);

    obj[1] = 99;
    obj.display();

    return 0;
}
#include <iostream>
using namespace std;

class Array
{
    int a[5];

public:
    Array(int r[5])
    {
        for (int i = 0; i < 5; i++)
            a[i] = r[i];
    }

    int &operator[](int index)
    {
        return a[index];
    }
};

int main()
{
    system("CLS");
    int x[5] = {1, 2, 3, 4, 5};
    Array obj(x);

    obj[2] = 99;

    for (int i = 0; i < 5; i++)
        cout << obj[i] << "\t";
    return 0;
}
