#include <iostream>
using namespace std;

class example
{
    int a, b;

public:
    void input()
    {
        cout << "Enter a and b: " << endl;
        cin >> a >> b;
    }

    void operator-()
    {
        a = -a;
        b = -b;
    }

    void display()
    {
        cout << "a = " << a << endl
             << "b = " << b << endl;
    }
};

int main()
{
    system("CLS");

    example e;
    e.input();
    cout << "Before Operator Overloading: " << endl;
    e.display();
    cout << "After Operator Overloading: " << endl;
    -e; // minus should be before object.
    e.display();

    return 0;
}
