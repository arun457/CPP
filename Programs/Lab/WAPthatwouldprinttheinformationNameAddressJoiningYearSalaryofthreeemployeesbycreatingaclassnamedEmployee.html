// WAP that would print the information (Name, Address ,Joining_Year, Salary) of three
// employees by creating a class named 'Employee'.

#include <iostream>
#define size 4 //Keep the size 1 greater than given in question
using namespace std;

class Employee
{
    string name, address;
    int joining_year;
    float salary;

public:
    void input()
    {
        cout << "Enter the name :";
        cin >> name;
        cout << "Enter the joining year:";
        cin >> joining_year;
        cout << "Enter Address: ";
        cin >> address;
        cout << "Enter the salary: ";
        cin >> salary;
    }

    void display()
    {
        cout << name << "\t" << joining_year << "\t" << address << "\t" << salary << endl;
    }
};

int main()
{
    system("CLS");
    int i;
    Employee e[size];
    cout << "Enter the name, joining year, Address and salary for " << size << " Employees" << endl;
    for (i = 1; i < size; i++)
    {
        e[i].input();
    }

    cout << "Name\tYear\tAddress\tSalary" << endl;
    for (i = 1; i < size; i++)
    {
        e[i].display();
    }
    return 0;
}
