//WAP with class named Book, which will represent title, author, price , publisher. Use 
//constructor for initialization of an objects and use dynamic memory allocation for string 
//type member and destructor in a proper way

#include <iostream>
using namespace std;

class Book
{
    string *title, *author, *publisher;
    float price;

public:
    Book(string title_init, string author_init, string publisher_init, float price_init)
    {
        title = new string;
        *title = title_init;
        author = new string;
        *author = author_init;
        publisher = new string;
        *publisher = publisher_init;
        price = price_init;
    }

    void display_data()
    {
        cout << "Title\tAuthor\tPublisher\tPrice" << endl;
        cout << *title << "\t" << *author << "\t" << *publisher << "\t" << price;
    }

    ~Book()
    {
        delete title;
        delete author;
        delete publisher;
    }
};

int main()
{
    system("CLS");
    Book B("Revolution 2020", "Chetan Bhagat", "Books Store India ptv.LTD", 800);
    B.display_data();
    return 0;
}

#include <iostream>
#include <string.h>
using namespace std;

class book
{
    char *title;
    char *author;
    char *publisher;
    float price;

public:
    book()
    {
        int length = 0;
        title = new char[length + 1];
        author = new char[length + 1];
        publisher = new char[length + 1];
        price = 0;
    }

    book(char *t, char *a, char *p, float pl)
    {
        int length;
        length = strlen(t);
        title = new char[length];
        strcpy(title, t);
        length = strlen(a);
        author = new char[length];
        strcpy(author, a);
        length = strlen(p);
        publisher = new char[length];
        strcpy(publisher, p);
        price = pl;
    }

    void display()
    {
        cout << "\nTitle of book: " << title;
        cout << "\nAuthor of book:" << author;
        cout << "\nPublisher of book:" << publisher;
        cout << "\nPrice of book:" << price;
    }

    ~book()
    {
        delete[] author;
        delete[] title;
        delete[] publisher;
    }
};
