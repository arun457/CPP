// WAP print the sum, difference and product of two complex numbers by creating a class
// named 'Complex' with separate functions for each operation whose real and imaginary
// parts are entered by the user

#include <iostream>
using namespace std;

class Complex
{
    float real, imaginary;

public:
    void input()
    {
        cout << "Enter the real and imaginary part of complex number: ";
        cin >> real >> imaginary;
    }

    Complex add(Complex c)
    {
        Complex temp;
        temp.real = real + c.real;
        temp.imaginary = imaginary + c.imaginary;
        return temp;
    }

    Complex subtract(Complex c)
    {
        Complex temp;
        temp.real = real - c.real;
        temp.imaginary = imaginary - c.imaginary;
        return temp;
    }

    Complex product(Complex c)
    {
        Complex temp;
        temp.real = (real * c.real) - (imaginary * c.imaginary);
        temp.imaginary = (real * c.imaginary) + (imaginary * c.real);
        return temp;
    }

    void display_sum()
    {
        if (imaginary >= 0)
        {
            cout << "The sum of the given Complex numbers are: " << real << " + " << imaginary << "i" << endl;
        }
        else
        {
            cout << "The sum of the given Complex numbers are: " << real << imaginary << "i" << endl;
        }
    }

    void display_difference()
    {
        if (imaginary >= 0)
        {
            cout << "The difference of the given Complex numbers are: " << real << " + " << imaginary << "i" << endl;
        }
        else
        {
            cout << "The difference of the given Complex numbers are: " << real << imaginary << "i" << endl;
        }
    }

    void display_product()
    {
        if (imaginary >= 0)
        {
            cout << "The product of the given Complex numbers are: " << real << " + " << imaginary << "i" << endl;
        }
        else
        {
            cout << "The product of the given Complex numbers are: " << real << imaginary << "i" << endl;
        }
    }
};

int main()
{
    system("CLS");
    Complex c1, c2, c3, c4, c5;

    // Taking input
    c1.input();
    c2.input();

    // Doing Operations
    c3 = c1.add(c2);
    c4 = c1.subtract(c2);
    c5 = c1.product(c2);

    // Displaying the results
    c3.display_sum();
    c4.display_difference();
    c5.display_product();

    return 0;
}
