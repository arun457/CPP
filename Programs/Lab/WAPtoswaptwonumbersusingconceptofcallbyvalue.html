#include <iostream>
using namespace std;

int a, b;

void swap_value()
{
    int z;
    z = a;
    a = b;
    b = z;
}

int main()
{
    system("CLS");
    cout << "Enter the value of a and b: ";
    cin >> a >> b;
    swap_value();
    cout << "The values of a and b after swapping is: " << a << " " << b;
}
