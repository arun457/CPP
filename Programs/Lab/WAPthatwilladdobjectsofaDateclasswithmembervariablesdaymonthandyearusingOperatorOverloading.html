// WAP that will add objects of a Date class with member variables day ,month and year using Operator Overloading.

#include<iostream>
using namespace std;

class Date
{
    int year,month,day;

public:
    Date()
    {

    }

    Date(int x,int y,int z):year(x),month(y),day(z)
    {

    }

    Date operator +(Date d)
    {
        Date temp;
        temp.day = day + d.day;

        temp.month = month + d.month + temp.day/30;
        temp.day = temp.day %30;

        temp.year =  year + d.year + temp.month / 12;
        temp.month = temp.month %12;

        return temp;
    }

    void display()
    {
        cout<<"The summed Date is :"<<year<<" Year\t"<<month<<" month\t"<<day<<" day";
    }
};

int main()
{
    Date d1(2058,1,20),d2(21,2,6),d3;
    d3 = d1 + d2;
    d3.display();
    return 0;
}
