#include <iostream>
using namespace std;

bool can_proceed = true;
static int timeindex = 1;
int sec_of_t1, sec_of_t2;

class Time
{
    int hr, minute, sec;

public:
    void input()
    {
        cout << "Enter the " << timeindex << "th time in hr, min, sec format: ";
        cin >> hr >> minute >> sec;
        timeindex = timeindex + 1;
    }

    void check_validation(Time t)
    {
        sec_of_t1 = (hr * 3600) + (minute * 60) + sec;
        sec_of_t2 = (t.hr * 3600) + (t.minute * 60) + t.sec;

        if (sec_of_t1 < sec_of_t2)
        {
            cout << "You have entered t1<t2" << endl
                 << "Please Enter t1>t2" << endl;
            can_proceed = false;
        }
    }

    Time add(Time t)
    {
        Time temp;

        temp.sec = sec_of_t1 + sec_of_t2;

        temp.minute = temp.sec / 60;
        temp.sec = temp.sec % 60;
        temp.hr = temp.minute / 60;
        temp.minute = temp.minute % 60;

        return temp;
    }

    Time subtract(Time t)
    {
        Time temp;

        temp.sec = sec_of_t1 - sec_of_t2;

        temp.minute = temp.sec / 60;
        temp.sec = temp.sec % 60;
        temp.hr = temp.minute / 60;
        temp.minute = temp.minute % 60;

        return temp;
    }

    void display_sum()
    {
        cout << "The sum of two time is: " << hr << " hr\t" << minute << " min\t" << sec << " sec" << endl;
    }

    void display_difference()
    {
        cout << "The Difference of two time is: " << hr << " hr\t" << minute << " min\t" << sec << " sec" << endl;
    }
};

int main()
{
    system("CLS");
    Time t1, t2, t3, t4;
    cout << "Enter the time (t1>t2) " << endl;
    t1.input();
    t2.input();
    t1.check_validation(t2);

    if (can_proceed != false)
    {
        t3 = t1.add(t2);
        t4 = t1.subtract(t2);
        t3.display_sum();
        t4.display_difference();
    }
    else
    {
        cout << "Exitting the Program";
    }
    return 0;
}
