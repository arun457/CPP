// Create a class mdistance to store the values in meter and centimeter and class edistance to store
// values in feet and inches. Perform addition of object of mdistance and object of edistance by using
// friend function.

#include <iostream>
using namespace std;

class edistance;
class mdistance
{
    float m, cm;

public:
    mdistance() : m(0), cm(0)
    {
    }

    mdistance(int x, int y) : m(x), cm(y)
    {
    }

    void display();

    friend mdistance add(mdistance d1, edistance d2);
};

void mdistance::display()
{
    cout << "The sum is: " << m << " m\t" << cm << " cm";
}

class edistance
{
    float ft, in;

public:
    edistance(int x, int y) : ft(x), in(y)
    {
    }

    friend mdistance add(mdistance d1, edistance d2);
};

mdistance add(mdistance d1, edistance d2)
{
    mdistance temp;

    temp.cm = d1.m * 100 + d1.cm + d2.ft * 12 * 2.54 + d2.in * 2.54;
    temp.m = (int)temp.cm / 100;
    temp.cm = temp.cm - (temp.m * 100);

    return temp;
}

int main()
{
    system("CLS");

    mdistance d1(1, 2), d3;
    edistance d2(3, 4);

    d3 = add(d1, d2);
    d3.display();

    return 0;
}
