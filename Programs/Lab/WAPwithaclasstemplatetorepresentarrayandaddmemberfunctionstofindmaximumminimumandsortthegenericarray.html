// WAP with a class template to represent array and add member functions to find maximum,
// minimum and sort the generic array

#include <iostream>
using namespace std;

template <class T, int Size>
class Array
{
    T a[Size];
    T maximum, minimum;

public:
    Array(T receive_array[Size])
    {
        for (int i = 0; i < Size; i++)
        {
            a[i] = receive_array[i];
        }
    }

    T Sort()
    {
        for (int i = 0; i < Size; i++)
        {
            for (int j = 0; j < Size; j++)
            {
                if (a[i] < a[j])
                {
                    T temp;
                    temp = a[i];
                    a[i] = a[j];
                    a[j] = temp;
                }
            }
        }

        maximum = a[Size - 1];
        minimum = a[0];
    }

    void display()
    {
        cout << "The Sorted Array in Ascending Order is: " << endl;
        for (int i = 0; i < Size; i++)
        {
            cout << a[i] << "\t";
        }
        cout << endl;
        cout << "Maximum = " << maximum << endl;
        cout << "Minimum = " << minimum << endl;
    }
};

int main()
{
    int init_array[5] = {5, 3, 9, 8, 4};
    Array<int, 5> a(init_array);
    a.Sort();
    a.display();
}
