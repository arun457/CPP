// Write a temperature conversion program that gives the user the option of converting
// degree Fahrenheit to degree Celsius or vice versa using the concept of class and object.

#include <iostream>
using namespace std;

class Temperature
{
    float tp;

public:
    void celcius()
    {
        cout << "Enter the temperature in Celcius:";
        cin >> tp;
        tp = (tp * 1.8) + 32;
        cout << "The temperature in Farenheit is: " << tp << " Degree farenheit";
    }

    void farenheit()
    {
        cout << "Enter the temperature in Farenheit:";
        cin >> tp;
        tp = (tp - 32) / 1.8;
        cout << "The temperature in Celcius is: " << tp << " Degree celcius";
    }
};

int main()
{
    system("CLS");
    int choice;
    Temperature t;
    cout << "Enter 1 for Celcius to Farenheit Conversion." << endl;
    cout << "Enter 2 for Farenheit to Celcius Conversion." << endl;
    cin >> choice;

    switch (choice)
    {
    case 1:
        t.celcius();
        break;
    case 2:
        t.farenheit();
        break;
    default:
        cout << "Invalid input.";
        break;
    }
    return 0;
}
