#include <iostream>
#include <math.h>
#define pi 3.14

using namespace std;

float area(float length = 0, float breadth = 0, float rhombus_height = 0, float radius = 0)
{
    if (radius != 0)
    {
        cout << "Circle" << endl;
        return (pi * pow(radius, 2));
    }
    else if (breadth != 0)
    {
        cout << "Rectangle" << endl;
        return (length * breadth);
    }
    else if (rhombus_height != 0)
    {
        cout << "Rhombus" << endl;
        return (length * rhombus_height);
    }
    else
    {
        cout << "Square" << endl;
        return (pow(length, 2));
    }
}

int main()
{
    system("CLS");
    float radius, length, breadth, rhombus_height;
    // For circle
    cout << "Enter the radius of circle: ";
    cin >> radius;
    cout << "The area of circle is: " << area(0, 0, 0, radius) << endl;

    // For Square
    cout << "Enter the lentgh of Square: ";
    cin >> length;
    cout << "The area of Square is: " << area(length) << endl;

    // For Rectangle
    cout << "Enter the lentgh and breadth of Rectangle: ";
    cin >> length >> breadth;
    cout << "The area of Rectangle is: " << area(length, breadth) << endl;

    // For Rhombus
    cout << "Enter the length and Height of Rhombus: ";
    cin >> length >> rhombus_height;
    cout << "The area of circle is: " << area(length, 0, rhombus_height) << endl;
}
