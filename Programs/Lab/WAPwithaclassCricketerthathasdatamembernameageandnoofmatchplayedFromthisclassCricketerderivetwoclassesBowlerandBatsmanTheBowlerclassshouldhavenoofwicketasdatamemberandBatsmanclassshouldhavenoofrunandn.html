// WAP with a class Cricketer that has data member name, age and no_of_match_played.
// From this class Cricketer derive two classes Bowler and Batsman. The Bowler class should have
// no_of_wicket as data member and Batsman class should have no_of_run and no_of_centuries as
// data members. Use appropriate member functions in all classes to make the program meaningful.

#include <iostream>
using namespace std;

class Cricketer
{
protected:
    string name;
    int age, no_of_match_played;

public:
    void input()
    {
        cout << "Name: ";
        cin >> name;
        cout << "Age: ";
        cin >> age;
        cout << "No. of Matches Played: ";
        cin >> no_of_match_played;
    }

    void display()
    {
        cout << "Name: " << name << endl;
        cout << "Age: " << age << endl;
        cout << "No. of Match Played: " << no_of_match_played << endl;
    }
};

class Bowler : public Cricketer
{
    int no_of_wickets;

public:
    Bowler()
    {
        cout << "For Bowler: " << endl;
        Cricketer::input();
        cout << "No. of Wickets: ";
        cin >> no_of_wickets;
    }

    void display()
    {
        cout << "For Bowler: " << endl;
        Cricketer::display();
        cout << "No. of Wickets: " << no_of_wickets << endl;
    }
};

class Batsman : public Cricketer
{
    int no_of_run, no_of_centuries;

public:
    Batsman()
    {
        cout << "For Batsman: " << endl;
        Cricketer::input();
        cout << "Enter the no of run: ";
        cin >> no_of_run;
        cout << "Enter the no of Centuries: ";
        cin >> no_of_centuries;
    }

    void display()
    {
        cout << "For Batsman: " << endl;
        Cricketer::display();
        cout << "No. of Run: " << no_of_run << endl;
        cout << "No. of Centuries: " << no_of_centuries << endl;
    }
};

int main()
{
    system("CLS");
    cout << "Enter the Details" << endl;
    Bowler bowler;
    cout << endl;
    Batsman batsman;
    cout << endl;

    cout << "The Entered Details are: " << endl;
    bowler.display();
    cout << endl;
    batsman.display();
}
