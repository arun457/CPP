// Write a program to create a class LandMeasure that reads Ropani, Ana, Paisa and
// Dam as data members. Write a function to pass two objects of type Land Measure and
// return their sum. (16 Ana = 1 Ropani, 4 Paisa: 1 Ana,4 Dam: 1 Paisa) using the concept of operator overloading.

#include <iostream>
using namespace std;

class LandMeasure
{
    int Ropani, Ana, Paisa, Dam;

public:
    LandMeasure()
    {
    }

    LandMeasure(int r, int a, int p, int d)
    {
        Ropani = r;
        Ana = a;
        Paisa = p;
        Dam = d;
    }

    LandMeasure operator+(LandMeasure L)
    {
        LandMeasure temp;

        temp.Dam = Dam + L.Dam;

        temp.Paisa = Paisa + L.Paisa + temp.Dam / 4;
        temp.Dam = temp.Dam % 4;

        temp.Ana = Ana + L.Ana + temp.Paisa / 4;
        temp.Paisa = temp.Paisa % 4;

        temp.Ropani = Ropani + L.Ropani + temp.Ana / 16;
        temp.Ana = temp.Ana % 16;

        return temp;
    }

    void Display()
    {
        cout << "Result: " << endl;
        cout << "Ropani: " << Ropani << endl;
        cout << "Ana: " << Ana << endl;
        cout << "Paisa: " << Paisa << endl;
        cout << "Dam: " << Dam << endl;
    }
};

int main()
{
    system("CLS");

    LandMeasure L1(1, 2, 3, 4), L2(5, 6, 7, 8), L3;

    L3 = L1 + L2;

    L3.Display();

    return 0;
}
