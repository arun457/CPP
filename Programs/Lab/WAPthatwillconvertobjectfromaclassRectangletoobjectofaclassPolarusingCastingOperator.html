// WAP that will convert object from a class Rectangle to object of a class Polar using Casting Operator

#include<iostream.>
#include<math.h>
#define pi 3.14
using namespace std;

class Polar
{
    float Theta,r;

public:
    Polar()
    {

    }

    void display()
    {
        cout<<"The coordinates in Polar form is: "<<r<<"\t"<<Theta<<endl;
    }

    friend class Rectangle;
};

class Rectangle
{
    float x,y;

public:
    Rectangle(int m, int n):x(m),y(n)
    {

    }

    operator Polar ()
    {
      Polar temp;

      temp.r = sqrt(pow(x,2)+pow(y,2));
      temp.Theta = atan(y/x);
      temp.Theta = 180/pi*temp.Theta;

      return temp;
    }
};


int main()
{
    Rectangle r(2,3);
    Polar p;

    p = r;
    p.display();
}
// WAP that will convert object from a class Rectangle to object of a class Polar using Casting Operator

#include <iostream.>
#include <math.h>
#define pi 3.14
using namespace std;

class Polar
{
    float Theta, r;

public:
    Polar()
    {
    }

    Polar(float m, float n) : r(m), Theta(n)
    {
    }

    void display()
    {
        cout << "The coordinates in Polar form is: " << r << "\t" << Theta << endl;
    }
};

class Rectangle
{
    float x, y;

public:
    Rectangle(int m, int n) : x(m), y(n)
    {
    }

    operator Polar()
    {
        return Polar(sqrt(pow(x, 2) + pow(y, 2)), atan(y / x) * 180 / pi);
    }
};

int main()
{
    system("CLS");
    Rectangle r(2, 3);
    Polar p;

    p = r;
    p.display();
}
