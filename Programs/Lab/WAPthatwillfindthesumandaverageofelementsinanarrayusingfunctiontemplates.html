// WAP that will find the sum and average of elements in an array using function templates

#include <iostream>
using namespace std;

template <class T>
T sum(T a[], int Size)
{
    T sum = 0;
    for (int i = 0; i < Size; i++)
    {
        sum = sum + a[i];
    }
    return (sum);
}

int main()
{
    int Size = 5;
    int a[Size] = {1, 2, 3, 4, 5};
    cout << "Sum = " << sum(a, Size) << endl;
    cout << "Average = " << sum(a, Size) / Size;
}
