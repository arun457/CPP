// WAP to convert currency from ruppee to dollar using both concept i.e. using constructor for
// data conversion as well as casting operator for data conversion.

#include <iostream>
#define rate 120
using namespace std;

class Rupee
{
    float r;

public:
    Rupee(int x) : r(x)
    {
    }

    friend class Dollar;
};

class Dollar
{
    float d;

public:
    Dollar() : d(0)
    {
    }

    Dollar(Rupee r1)
    {
        d = r1.r / rate;
    }

    void display()
    {
        cout << "The Money in Dollar is : " << d << " Dollar" << endl;
    }
};

int main()
{
    system("CLS");
    Rupee rupee(240);
    Dollar dollar;

    dollar = rupee;
    dollar.display();

    return 0;
}
// WAP to convert currency from ruppee to dollar using both concept i.e. using constructor for
// data conversion as well as casting operator for data conversion.

#include <iostream>
#define rate 120
using namespace std;

class Dollar
{
    float d;

public:
    Dollar() : d(0)
    {
    }

    void display()
    {
        cout << "The Money in Dollar is : " << d << " Dollar" << endl;
    }

    friend class Rupee;
};

class Rupee
{
    float r;

public:
    Rupee(int x) : r(x)
    {
    }

    operator Dollar()
    {
        Dollar temp;
        temp.d = r / rate;
        return temp;
    }
};

int main()
{
    system("CLS");
    Rupee rupee(240);
    Dollar dollar;

    dollar = rupee;
    dollar.display();

    return 0;
}
