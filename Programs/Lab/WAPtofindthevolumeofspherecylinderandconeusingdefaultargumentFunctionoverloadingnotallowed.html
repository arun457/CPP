#include <iostream>
#include <math.h>
#define pi 3.14
using namespace std;

float volume(float radius = 0, float cylinder_height = 0, float cone_height = 0)
{
    if (cylinder_height == 0 && cone_height == 0)
    {
        cout << "Sphere" << endl;
        return ((4 / 3) * pi * pow(radius, 3));
    }
    else if (cone_height == 0)
    {
        cout << "cylinder" << endl;
        return (pi * pow(radius, 2) * cylinder_height);
    }
    else if (cylinder_height == 0)
    {
        cout << "cone" << endl;
        return ((pi * pow(radius, 2) * cone_height) / 3);
    }
}

int main()
{
    system("CLS");
    float radius, cylinder_height, cone_height;
    cout << "Enter the radius of sphere: ";
    cin >> radius;
    cout << volume(radius) << endl;
    cout << "Enter the radius and height of cylinder: ";
    cin >> radius >> cylinder_height;
    cout << volume(radius, cylinder_height) << endl;
    cout << "Enter the radius and height of cone: ";
    cin >> radius >> cone_height;
    cout << volume(radius, cone_height) << endl;
}
