//WAP to concatenate two user given string using the concept of Operator Overloading.

#include<iostream>
#include<string.h>
using namespace std;

class String
{
    char *s;
public:
    void input()
    {
        cout<<"Enter the string"<<endl;
        s = new char[20];
        cin>>s;

    }

    String operator +(String s3)
    {
        String temp;
        int len = strlen(s)+strlen(s3.s);
        temp.s = new char[len];
        temp.s = strcat(s,s3.s);
        return temp;
    }

    void display()
    {
        cout<<"The concatenated string is: "<<s<<endl;
    }
};

int main()
{
    String s1,s2,s4;
    s1.input();
    s2.input();
    s4 = s1+s2;
    s4.display();
}
