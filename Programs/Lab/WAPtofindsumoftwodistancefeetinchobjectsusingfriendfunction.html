// WAP to find sum of two distance (feet, inch) objects using friend function.

#include <iostream>
using namespace std;

class Distance
{
    int ft;
    int in;

public:
    Distance()
    {
    }

    Distance(int f, int i)
    {
        ft = f;
        in = i;
    }

    void display()
    {
        cout << "The sum is: " << ft << " feet\t" << in << "inch";
    }

    friend Distance add(Distance d1, Distance d2);
};

Distance add(Distance d1, Distance d2)
{
    Distance temp;
    temp.in = d1.in + d2.in;
    temp.ft = d1.ft + d2.ft + temp.in / 12;
    temp.in = temp.in % 12;
    return temp;
}

int main()
{
    system("CLS");
    Distance d1(1, 11), d2(2, 3), d3;
    d3 = add(d1, d2);
    d3.display();
    return 0;
}
// WAP to find sum of two distance (feet, inch) objects using friend function.

#include <iostream>
using namespace std;

class Distance
{
    int ft;
    int in;

public:
    Distance() : ft(0), in(0)
    {
    }

    Distance(int f, int i)
    {
        ft = f;
        in = i;
    }

    void display()
    {
        cout << "The sum is: " << ft << " feet\t" << in << "inch";
    }

    friend Distance add(Distance d1, Distance d2);
};

Distance add(Distance d1, Distance d2)
{
    Distance temp;
    temp.in = d1.ft * 12 + d1.in + d2.ft * 12 + d2.in;
    temp.ft = temp.in / 12;
    temp.in = temp.in % 12;
    return temp;
}

int main()
{
    system("CLS");
    Distance d1(1, 11), d2(2, 3), d3;
    d3 = add(d1, d2);
    d3.display();
    return 0;
}
