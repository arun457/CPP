// WAP with two base classes “Employee” and “Student” and a third class ‘Manager’
// inherited from those two base classes. Assume that both of base classes has data members
//‘name’ and ‘address’. Add suitable member function and write a meaningful program. Also
// illustrate the concept of function overriding

#include <iostream>
using namespace std;

class Employee
{
protected:
    string name, address;
    int employee_id;

    void display()
    {
    }
};

class Student
{
protected:
    string name, address;
    int student_id;

    void display()
    {
    }
};

class Manager : public Employee, public Student
{
public:
    Manager()
    {
        cout << "Enter the Name, Address, Employee_id and Student_id: ";
        cin >> Employee::name >> Employee::address >> employee_id >> student_id;
    }

    void display()
    {
        cout << "Name\tAddress\tEmployee_id\tStudent_id" << endl;
        cout << Employee::name << "\t" << Employee::address << "\t" << employee_id << "\t" << student_id << endl;
    }
};

int main()
{
    Manager mgr;
    mgr.display();
    return 0;
}
