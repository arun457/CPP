#include <iostream>
using namespace std;

int main()
{
    system("CLS");
    int i, n, a, b, c;
    cout << "Enter the number of terms of fibonacci numbers you want: ";
    cin >> n;
    a = 0;
    b = 1;
    cout << a << "\t";
    cout << b << "\t";

    for (i = 2; i < n; i++)
    {
        c = a + b;
        cout << c << "\t";
        a = b;
        b = c;
    }
}
