// Define a class named Course. Derive three classes from this class named: Mathematics,
// Science and Engineering.Then, derive two classes from Science named : Physics and Chemistry.Define data member and member functions as appropriate.Also illustrate the concept of function
// overriding and accessing overridden member from the derived class in your program

#include <iostream>
using namespace std;

class Course
{
protected:
    int course_id;
    int total_marks, theory_mark;

public:
    void input()
    {
        cout << "Enter course_id: ";
        cin >> course_id;
        cout << "Enter total marks: ";
        cin >> total_marks;
        cout << "Enter theory mark: ";
        cin >> theory_mark;
    }

    void display()
    {
        cout << "The details are: " << endl;
        cout << "Course id = " << course_id << endl;
        cout << "Total Mark = " << total_marks << endl;
        cout << "Theory Mark = " << theory_mark << endl;
    }
};

class Mathematics : public Course
{
    int Thesis_mark;

public:
    void input()
    {
        cout << "For Mathematics: " << endl;
        Course::input();
        cout << "Enter thesis mark: ";
        cin >> Thesis_mark;
    }

    void display()
    {
        cout << "For Mathematics: " << endl;
        Course::display();
        cout << "Thesis Mark = " << Thesis_mark << endl;
    }
};

class Science : public Course
{
    int total_practical_mark;

public:
    void input()
    {
        Course::input();
        cout << "Enter Total Practical mark: ";
        cin >> total_practical_mark;
    }

    void display()
    {
        Course::display();
        cout << "Total Practical Mark = " << total_practical_mark << endl;
    }
};
