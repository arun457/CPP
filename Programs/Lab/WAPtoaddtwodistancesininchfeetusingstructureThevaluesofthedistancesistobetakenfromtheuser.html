#include <iostream>
using namespace std;

struct distance
{
    int feet;
    int inch;
};

int main()
{
    system("ClS");
    struct distance d1, d2, result;
    cout << "Enter the first distance in inch and ft: ";
    cin >> d1.feet >> d1.inch;

    cout << "Enter the Second distance in inch and ft: ";
    cin >> d2.feet >> d2.inch;

    result.feet = d1.feet + d2.feet;
    result.inch = d1.inch + d2.inch;
    if (result.inch >= 12)
    {
        result.feet = result.feet + (result.inch / 12);
        result.inch = result.inch - ((result.inch / 12) * 12);
    }

    cout << "The sum of distance is: " << result.feet << " feet " << result.inch << " inch";
}
