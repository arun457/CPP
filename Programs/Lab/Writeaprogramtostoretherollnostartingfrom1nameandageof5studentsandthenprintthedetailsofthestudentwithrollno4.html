#include <iostream>
using namespace std;

struct Student
{
    int roll_no;
    string name;
    int age;
};

int main()
{
    int n = 5;
    bool roll_no_4_found = false;
    system("CLS");
    struct Student s[6];
    int i;
    cout << "Enter the name, age and roll_no of 5 students: ";
    for (i = 1; i <= n; i++)
    {
        cin >> s[i].name >> s[i].age >> s[i].roll_no;
    }

    for (i = 1; i <= n; i++)
    {
        if (s[i].roll_no == 4)
        {
            cout << "The name, age and roll_no of student having roll no = 4 is: " << s[i].name << " " << s[i].age << " " << s[i].roll_no;
            bool roll_no_4_found = true;
        }
    }

    if (roll_no_4_found == false)
    {
        cout << "There is no Student with roll no = 4.";
    }
}
