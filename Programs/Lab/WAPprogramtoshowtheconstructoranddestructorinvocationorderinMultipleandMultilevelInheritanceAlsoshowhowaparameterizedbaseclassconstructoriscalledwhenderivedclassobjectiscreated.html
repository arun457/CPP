// WAP program to show the constructor and destructor invocation order in Multiple and
// Multilevel Inheritance. Also show how a parameterized base class constructor is called when
// derived class object is created.

#include <iostream>
using namespace std;

class Base1
{
public:
    Base1()
    {
        cout << "Base1 Constructor Called." << endl;
    }

    ~Base1()
    {
        cout << "Base1 Destructor Called." << endl;
    }
};

class Base2
{
public:
    Base2()
    {
        cout << "Base2 Constructor Called." << endl;
    }

    ~Base2()
    {
        cout << "Base2 Destructor Called." << endl;
    }
};

class Derived : public Base1, public Base2
{
public:
    Derived()
    {
        cout << "Derived Constructor Called." << endl;
    }

    ~Derived()
    {
        cout << "Derived Destructor Called." << endl;
    }
};

int main()
{
    system("CLS");
    Derived obj;
}

// WAP program to show the constructor and destructor invocation order in Multiple and
// Multilevel Inheritance. Also show how a parameterized base class constructor is called when
// derived class object is created.

#include <iostream>
using namespace std;

class Base
{
    int x;

public:
    Base(int m) : x(m)
    {
    }

    void display()
    {
        cout << "X = " << x << endl;
    }
};

class Derived : public Base
{
    int y;

public:
    Derived(int m, int n) : y(n), Base(m)
    {
    }

    void display()
    {
        Base::display();
        cout << "Y = " << y << endl;
    }
};

int main()
{
    system("CLS");
    Derived obj(1, 2);
    obj.display();
    return 0;
}
