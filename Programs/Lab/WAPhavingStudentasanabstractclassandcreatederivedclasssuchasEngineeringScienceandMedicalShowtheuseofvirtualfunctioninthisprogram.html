// WAP having Student as an abstract class and create derived class such as Engineering,
// Science and Medical. Show the use of virtual function in this program

#include <iostream>
using namespace std;

class Student
{
protected:
    int no_of_stds;

    virtual int no_of_students() = 0;
};

class Engineering : public Student
{
public:
    Engineering(int n)
    {
        no_of_stds = n;
    }

    int no_of_students()
    {
        return (no_of_stds);
    }
};

class Medical : public Student
{
public:
    Medical(int n)
    {
        no_of_stds = n;
    }

    int no_of_students()
    {
        return (no_of_stds);
    }
};

int main()
{
    Engineering e(5);
    Medical m(10);
    cout << "The number of Students in Engineering is: " << e.no_of_students() << endl;
    cout << "The number of Students in Medical is: " << m.no_of_students() << endl;
}
