// WAP which depicts the concept of Hierarchical and Hybrid inheritance .Use appropriate
// classes along with its data members and member functions

#include <iostream>
using namespace std;

class Side
{
protected:
    int x;

public:
    void set_value(int i)
    {
        x = i;
    }
};

class Square : public Side
{
public:
    int square()
    {
        return (x * x);
    }
};

class Cube : public Side
{
public:
    int cube()
    {
        return (x * x * x);
    }
};

int main()
{
    system("CLS");
    Square sq;
    Cube cu;

    sq.set_value(2);
    cu.set_value(3);

    cout << "The Square Number is: " << sq.square() << endl;
    cout << "The Cube Number is: " << cu.cube();
    return 0;
}
// WAP which depicts the concept of Hierarchical and Hybrid inheritance .Use appropriate
// classes along with its data members and member functions

#include <iostream>
using namespace std;

class Student
{
protected:
    int roll;

public:
    void get_roll()
    {
        cout << "Enter Roll_no: ";
        cin >> roll;
    }

    void display_roll()
    {
        cout << "Roll No: " << roll << endl;
    }
};

class Exam : virtual public Student
{
protected:
    int theory_mark;

public:
    void get_theory_mark()
    {
        cout << "Enter the theory mark: ";
        cin >> theory_mark;
    }

    void display_theory_mark()
    {
        cout << "Theory Mark: " << theory_mark << endl;
    }
};

class Sports : virtual public Student
{
protected:
    int sports_mark;

public:
    void get_sports_mark()
    {
        cout << "Enter Sports Mark: ";
        cin >> sports_mark;
    }

    void display_sports_mark()
    {
        cout << "Sports Mark: " << sports_mark << endl;
    }
};

class Result : public Exam, public Sports
{
    int total_mark;

public:
    void display()
    {
        total_mark = theory_mark + sports_mark;
        cout << "Total Marks = " << total_mark << endl;
    }
};

int main()
{
    system("CLS");
    Result s;
    s.get_roll();
    s.get_theory_mark();
    s.get_sports_mark();
    s.display();
    return 0;
}
