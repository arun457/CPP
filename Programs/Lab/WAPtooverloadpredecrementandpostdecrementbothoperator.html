// WAP to overload - - (pre decrement and post decrement both) operator.

#include <iostream>
using namespace std;

class Sample
{
    int x;

public:
    Sample(int y) : x(y)
    {
    }

    void operator--()
    {
        --x;
    }

    void operator--(int)
    {
        x--;
    }

    void display()
    {
        cout << "The value of x is: " << x << endl;
    }
};

int main()
{
    system("CLS");
    Sample s(5);
    --s;
    s.display();
    s--;
    s.display();
}
