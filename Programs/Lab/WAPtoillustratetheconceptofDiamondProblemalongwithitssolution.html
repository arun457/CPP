// WAP to illustrate the concept of Diamond Problem along with its solution.

#include <iostream>
using namespace std;

class A
{
protected:
    int a = 0;
};

class B : public virtual A
{
protected:
    int b = 1;
};

class C : virtual public A
{
protected:
    int c = 2;
};

class D : public B, public C
{
protected:
    int d = 3;

public:
    void display()
    {
        cout << "a\tb\tc\td" << endl;
        cout << a << "\t" << b << "\t" << c << "\t" << d << endl;
    }
};

int main()
{
    system("CLS");
    D obj;
    obj.display();
}
